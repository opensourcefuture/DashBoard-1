# 设置代理分组，
proxy-groups:
-
  # 组名
  name: Proxy
  # 行为，有多种，具体看教程，如自动选延迟低的、负载均衡啥的
  type: select
  use:
    # 把前面的机场节点放到组里面 前面proxy-providers下的机场名字
    - V2Yun
    - SpeederOne
    - BaiPiao
-
  # 国内流量分组
  name: Domestic
  type: select
  use:
    # 同上  机场得放在use下
    - V2Yun
  proxies:
    # 可以引用其他的组，但是得放在proxies下
    # DIRECT（直连）为自带的分组
    - DIRECT
    # 引用上一个分组
    - Proxy
-
  name: GooglePlay
  type: select
  proxies:
    - Domestic
    - Proxy
  use:
    - SpeederOne
-
  name: Drive
  type: select
  proxies:
    - Domestic
    - Proxy
  use:
    - V2Yun
    - SpeederOne
    - BaiPiao
-
  name: Tv
  type: select
  proxies:
    - Domestic
    - Proxy
  use:
    - V2Yun
    - SpeederOne
    - BaiPiao

  name: Others

  proxies:
    - Proxy

# 分流规则 使用规则，按顺序匹配
#
# 重点(敲黑板)
# 规则从上往下顺序匹配，前面匹配了就不会继续向后匹配
#
rules:
  # 防止回环
  - IP-CIDR,127.0.0.1/32,REJECT,no-resolve
  # RULE-SET使用上面的rule-providers的规则，语法：RULE-SET,规则名,匹配后的行为
  # 此处为直连，不走代理
  - RULE-SET,Local,DIRECT
  # 此处行为为拒绝，即没网
  - RULE-SET,Ad,REJECT
  - RULE-SET,Feeb,REJECT
  # 此处行为为使用GooglePlay分组（proxy-groups）
  - RULE-SET,Play,GooglePlay
  # 此处行为为使用Drive分组（proxy-groups）
  - RULE-SET,GoogleDrive,Drive
  - RULE-SET,Onedrive,Drive
  - RULE-SET,Telegram,Proxy
  - RULE-SET,Netflix,Tv
  - RULE-SET,Twitter,Tv
  - RULE-SET,YouTube,Tv
  - RULE-SET,FCM,Proxy
  - RULE-SET,oTv,Tv
  - RULE-SET,Apps,Proxy
  # 匹配国内的流量
  - GEOIP,CN,Domestic
  # MATCH为全匹配，意思就是其他没匹配到的走Others分组（proxy-groups）
  - MATCH,Others
